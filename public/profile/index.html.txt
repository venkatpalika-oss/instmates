<script type="module">
import { db } from "/assets/js/firebase.js";
import { doc, getDoc }
from "https://www.gstatic.com/firebasejs/9.23.0/firebase-firestore.js";

const container = document.getElementById("profileContainer");

const params = new URLSearchParams(window.location.search);
const uid = params.get("uid");

if (!uid) {
  container.innerHTML = "<p class='muted'>Profile not found.</p>";
} else {
  loadProfile(uid);
}

async function loadProfile(uid) {

  const snap = await getDoc(doc(db, "profiles", uid));

  if (!snap.exists()) {
    container.innerHTML = "<p class='muted'>Profile not found.</p>";
    return;
  }

  const p = snap.data();

  const skillsHTML = Array.isArray(p.skills) && p.skills.length
    ? `<div class="tags">
        ${p.skills.map(skill =>
          `<span class="tag">${escapeHTML(skill)}</span>`
        ).join("")}
       </div>`
    : "<p class='muted'>No skills added.</p>";

  const experienceHTML = p.experienceDetails
    ? `<div class="card" style="margin-top:15px;">
         <p>${escapeHTML(p.experienceDetails)}</p>
       </div>`
    : "<p class='muted'>No experience added.</p>";

  const troubleshootingHTML = p.majorTroubleshooting
    ? `<div class="card" style="margin-top:15px;">
         <p>${escapeHTML(p.majorTroubleshooting)}</p>
       </div>`
    : "<p class='muted'>No major troubleshooting added.</p>";

  container.innerHTML = `

  <!-- HERO SECTION -->
  <div style="text-align:center;padding:30px 0">

    <img src="${p.photoURL || "/assets/images/default-avatar.png"}"
         style="width:140px;height:140px;border-radius:50%;object-fit:cover;border:4px solid #fff;box-shadow:0 5px 20px rgba(0,0,0,0.15)">

    <h1 style="margin-top:15px">
      ${escapeHTML(p.fullName || "Technician")}
      ${p.isVerified
        ? `<span style="color:#0d6efd;font-size:18px;margin-left:8px">‚úî</span>`
        : ""
      }
    </h1>

    <p class="muted">${escapeHTML(p.role || "")}</p>
    <p class="muted">üìç ${escapeHTML(p.location || "")}</p>

    <a href="/message/?to=${uid}" class="btn btn-primary" style="margin-top:15px">
      Message
    </a>

  </div>

  <hr>

  <!-- ABOUT -->
  <h3>About</h3>
  <p>${escapeHTML(p.bio || "No description provided.")}</p>

  <hr>

  <!-- SKILLS -->
  <h3>Skills</h3>
  ${skillsHTML}

  <hr>

  <!-- EXPERIENCE -->
  <h3>Experience</h3>
  ${experienceHTML}

  <hr>

  <!-- MAJOR TROUBLESHOOTING -->
  <h3>Major Troubleshooting Case</h3>
  ${troubleshootingHTML}

  `;
}

function escapeHTML(str) {
  return String(str)
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;");
}
